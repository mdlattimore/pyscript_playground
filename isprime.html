Enter any whole number in the box and click 'submit' to learn whether it is a prime number




def is_prime(*args, **kwargs): num = int(Element('num').value) n = 2 while n <= num ** .5: if num % n == 0: Element('result').write(f"{num:,} is not a prime number") return False n += 1 Element('result').write(f"{num:,} is a prime number") return True
